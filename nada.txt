//src\types\clipboard.ts
import type { FileEntry } from "./files";

export type ClipboardMode = "copy" | "cut";

export interface ClipboardItem {
  item: FileEntry;
  mode: ClipboardMode;
  createdAt: number;
}

export type ClipboardState = ClipboardItem | null;

//src\services\clipboardService.ts
import type {
  ClipboardItem,
  ClipboardMode,
  ClipboardState,
} from "../types/clipboard";
import type { FileEntry } from "../types/files";
import { storageService } from "./storageService";

export const clipboardService = {
  get(): ClipboardState {
    return storageService.getClipboard();
  },

  set(item: FileEntry, mode: ClipboardMode): ClipboardItem {
    const clip: ClipboardItem = {
      item,
      mode,
      createdAt: Date.now(),
    };

    storageService.setClipboard(clip);
    return clip;
  },

  clear() {
    storageService.clearClipboard();
  },
};

//src\components\ClipboardBar.tsx
import React from "react";
import { IonButton } from "@ionic/react";
import type { ClipboardState } from "../types/clipboard";

interface Props {
  clipboard: ClipboardState;
  onPaste: () => void;
  onClear: () => void;
}

const ClipboardBar: React.FC<Props> = ({ clipboard, onPaste, onClear }) => {
  if (!clipboard) return null;

  return (
    <div className="px-4 py-2 text-sm opacity-90 flex items-center justify-between">
      <span>
        Portapapeles: <strong>{clipboard.item.name}</strong> (
        {clipboard.mode === "copy" ? "copiar" : "mover"})
      </span>
      <div className="flex gap-2">
        <IonButton size="small" onClick={onPaste}>Pegar</IonButton>
        <IonButton size="small" fill="outline" onClick={onClear}>Limpiar</IonButton>
      </div>
    </div>
  );
};

export default ClipboardBar;

//src\components\ActionBar.tsx
import React from "react";
import {
  IonButton,
  IonIcon,
  IonInput,
  IonItem,
  IonLabel,
  IonSelect,
  IonSelectOption,
} from "@ionic/react";
import { gridOutline, listOutline, checkmarkDoneOutline } from "ionicons/icons";
import type { FilterType, SortMode, ViewMode } from "../types/files";

interface Props {
  search: string;
  sortBy: SortMode;
  filterBy: FilterType;
  viewMode: ViewMode;
  selectionMode?: boolean;
  onSearchChange: (value: string) => void;
  onSortChange: (value: SortMode) => void;
  onFilterChange: (value: FilterType) => void;
  onViewModeChange: (value: ViewMode) => void;
  onToggleSelectionMode?: () => void;
}

const ActionBar: React.FC<Props> = ({
  search,
  sortBy,
  filterBy,
  viewMode,
  selectionMode,
  onSearchChange,
  onSortChange,
  onFilterChange,
  onViewModeChange,
  onToggleSelectionMode,
}) => {
  return (
    <div className="space-y-2">
      <IonItem>
        <IonInput
          placeholder="Buscar archivos..."
          value={search}
          onIonInput={(e) => onSearchChange(e.detail.value || "")}
        />
      </IonItem>

      <div className="grid grid-cols-2 gap-2">
        <IonItem>
          <IonLabel>Orden</IonLabel>
          <IonSelect
            value={sortBy}
            onIonChange={(e) => onSortChange(e.detail.value)}
          >
            <IonSelectOption value="name-asc">Nombre A-Z</IonSelectOption>
            <IonSelectOption value="name-desc">Nombre Z-A</IonSelectOption>
            <IonSelectOption value="date-desc">Más recientes</IonSelectOption>
            <IonSelectOption value="date-asc">Más antiguos</IonSelectOption>
            <IonSelectOption value="size-desc">Más grandes</IonSelectOption>
            <IonSelectOption value="size-asc">Más pequeños</IonSelectOption>
          </IonSelect>
        </IonItem>

        <IonItem>
          <IonLabel>Filtro</IonLabel>
          <IonSelect
            value={filterBy}
            onIonChange={(e) => onFilterChange(e.detail.value)}
          >
            <IonSelectOption value="all">Todos</IonSelectOption>
            <IonSelectOption value="folders">Carpetas</IonSelectOption>
            <IonSelectOption value="images">Imágenes</IonSelectOption>
            <IonSelectOption value="videos">Videos</IonSelectOption>
            <IonSelectOption value="audio">Audio</IonSelectOption>
            <IonSelectOption value="documents">Documentos</IonSelectOption>
            <IonSelectOption value="apk">APK</IonSelectOption>
            <IonSelectOption value="archives">Comprimidos</IonSelectOption>
          </IonSelect>
        </IonItem>
      </div>

      <div className="flex gap-2">
        <IonButton
          expand="block"
          fill={viewMode === "list" ? "solid" : "outline"}
          onClick={() => onViewModeChange("list")}
        >
          <IonIcon icon={listOutline} slot="start" />
          Lista
        </IonButton>

        <IonButton
          expand="block"
          fill={viewMode === "grid" ? "solid" : "outline"}
          onClick={() => onViewModeChange("grid")}
        >
          <IonIcon icon={gridOutline} slot="start" />
          Grilla
        </IonButton>

        {onToggleSelectionMode && (
          <IonButton expand="block" fill={selectionMode ? "solid" : "outline"} onClick={onToggleSelectionMode}>
            <IonIcon icon={checkmarkDoneOutline} slot="start" />
            Selección
          </IonButton>
        )}
      </div>
    </div>
  );
};

export default ActionBar;

//src/pages/Browser.tsx
import React, { useEffect, useMemo, useState } from "react";
import {
  IonPage,
  IonHeader,
  IonToolbar,
  IonTitle,
  IonContent,
  IonButton,
  IonIcon,
  IonToast,
  IonRefresher,
  IonRefresherContent,
  IonList,
  IonGrid,
  IonRow,
  IonCol,
  IonFab,
  IonFabButton,
  IonAlert,
  IonActionSheet,
} from "@ionic/react";
import {
  arrowBackOutline,
  addOutline,
  clipboardOutline,
  copyOutline,
  cutOutline,
  trashOutline,
  shareOutline,
  createOutline,
  informationCircleOutline,
  starOutline,
  star,
  closeOutline,
} from "ionicons/icons";
import { useLocation, useNavigate } from "react-router-dom";
import { clipboardService } from "../services/clipboardService";

import type {
  FileEntry,
  FilterType,
  SortMode,
  FileVisualType,
} from "../types/files";
import type { ClipboardMode } from "../types/clipboard";

import { safService } from "../services/safService";
import { storageService } from "../services/storageService";

import { filterFiles, sortFiles } from "../utils/filters";

import FileItem from "../components/FileItem";
import ActionBar from "../components/ActionBar";
import EmptyState from "../components/EmptyState";
import Breadcrumbs from "../components/Breadcrumbs";
import PropertiesModal from "../components/PropertiesModal";

interface BrowserState {
  uri?: string;
  name?: string;
  special?: "favorites" | "recents";
}

interface Crumb {
  name: string;
  uri: string;
}

const Browser: React.FC = () => {
  const navigate = useNavigate();
  const location = useLocation();
  const state = (location.state || {}) as BrowserState;

  const [currentUri, setCurrentUri] = useState(state.uri || "");
  const [currentName, setCurrentName] = useState(state.name || "Explorador");
  const [clipboard, setClipboardState] = useState(clipboardService.get());
  const [items, setItems] = useState<FileEntry[]>([]);
  const [loading, setLoading] = useState(true);

  const [search, setSearch] = useState("");
  const [sortBy, setSortBy] = useState<SortMode>("name-asc");
  const [filterBy, setFilterBy] = useState<FilterType>("all");
  const [viewMode, setViewMode] = useState<FileVisualType>(storageService.getViewMode());

  const [toast, setToast] = useState("");

  const [selectionMode, setSelectionMode] = useState(false);
  const [selected, setSelected] = useState<FileEntry[]>([]);

  const [activeItem, setActiveItem] = useState<FileEntry | null>(null);
  const [showActionSheet, setShowActionSheet] = useState(false);
  const [showProperties, setShowProperties] = useState(false);

  const [showCreateFolder, setShowCreateFolder] = useState(false);
  const [newFolderName, setNewFolderName] = useState("");

  const [showRename, setShowRename] = useState(false);
  const [renameValue, setRenameValue] = useState("");

  const [breadcrumbs, setBreadcrumbs] = useState<Crumb[]>([]);

  const isSpecialFavorites = state.special === "favorites";
  const isSpecialRecents = state.special === "recents";
  const isSpecialMode = isSpecialFavorites || isSpecialRecents;

  const loadDirectory = async (uri: string, name?: string, pushCrumb = true) => {
    try {
      setLoading(true);

      const result = await safService.listDirectory(uri);

      const favoriteUris = storageService.getFavorites().map((f) => f.uri);

      const normalized = result.items.map((item) => ({
        ...item,
        favorite: favoriteUris.includes(item.uri),
      }));

      setItems(normalized);
      setCurrentUri(result.currentUri);
      if (name) setCurrentName(name);

      if (pushCrumb) {
        setBreadcrumbs((prev) => {
          const existsIndex = prev.findIndex((c) => c.uri === result.currentUri);
          if (existsIndex >= 0) {
            return prev.slice(0, existsIndex + 1);
          }

          return [
            ...prev,
            {
              name: name || extractNameFromUri(result.currentUri),
              uri: result.currentUri,
            },
          ];
        });
      }
    } catch (error: any) {
      setToast(error?.message || "No se pudo abrir la carpeta");
    } finally {
      setLoading(false);
    }
  };

  const loadSpecial = () => {
    if (isSpecialFavorites) {
      const favorites = storageService.getFavorites().map((f) => ({
        ...f,
        favorite: true,
      }));
      setItems(favorites);
      setLoading(false);
      return;
    }

    if (isSpecialRecents) {
      const favoriteUris = storageService.getFavorites().map((f) => f.uri);
      const recents = storageService.getRecents().map((f) => ({
        ...f,
        favorite: favoriteUris.includes(f.uri),
      }));
      setItems(recents);
      setLoading(false);
      return;
    }
  };

  useEffect(() => {
    if (isSpecialMode) {
      loadSpecial();
      return;
    }

    if (state.uri) {
      setBreadcrumbs([
        {
          name: state.name || extractNameFromUri(state.uri),
          uri: state.uri,
        },
      ]);
      loadDirectory(state.uri, state.name, false);
    } else {
      setLoading(false);
    }
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, []);

  const visibleItems = useMemo(() => {
    return sortFiles(filterFiles(items, search, filterBy), sortBy);
  }, [items, search, filterBy, sortBy]);

  const refreshCurrent = async () => {
    if (isSpecialMode) {
      loadSpecial();
      return;
    }

    if (!currentUri) return;
    await loadDirectory(currentUri, currentName, false);
  };

  const handleBack = () => {
    if (selectionMode) {
      setSelectionMode(false);
      setSelected([]);
      return;
    }

    if (isSpecialMode) {
      navigate("/");
      return;
    }

    if (breadcrumbs.length > 1) {
      const newCrumbs = [...breadcrumbs];
      newCrumbs.pop();
      const prev = newCrumbs[newCrumbs.length - 1];
      setBreadcrumbs(newCrumbs);
      loadDirectory(prev.uri, prev.name, false);
      return;
    }

    navigate("/");
  };

  const handleItemClick = async (item: FileEntry) => {
    if (selectionMode) {
      toggleSelect(item);
      return;
    }

    if (item.isDirectory) {
      await loadDirectory(item.uri, item.name);
      return;
    }

    try {
      storageService.addRecent(item);
      await safService.openFile(item.uri, item.mimeType);
    } catch (error: any) {
      setToast(error?.message || "No se pudo abrir el archivo");
    }
  };

  const toggleSelect = (item: FileEntry) => {
    setSelected((prev) => {
      const exists = prev.some((x) => x.uri === item.uri);
      if (exists) return prev.filter((x) => x.uri !== item.uri);
      return [...prev, item];
    });
  };

  const toggleFavorite = (item: FileEntry) => {
    const isFav = storageService.toggleFavorite(item);

    setItems((prev) =>
      prev.map((x) =>
        x.uri === item.uri ? { ...x, favorite: isFav } : x
      )
    );

    setToast(isFav ? "Añadido a favoritos" : "Quitado de favoritos");
  };

  const handleOptions = (item: FileEntry) => {
    setActiveItem(item);
    setRenameValue(item.name);
    setShowActionSheet(true);
  };

  const handleDelete = async (item: FileEntry) => {
    try {
      await safService.delete(item.uri);
      setToast("Eliminado");
      refreshCurrent();
    } catch (error: any) {
      setToast(error?.message || "No se pudo eliminar");
    }
  };

  const handleRename = async (newNameParam?: string) => {
    if (!activeItem) return;
  
    const finalName = (newNameParam ?? renameValue).trim();
    if (!finalName) return;
  
    // si el nombre es exactamente el mismo, no hacemos nada
    if (finalName === activeItem.name.trim()) {
      setShowRename(false);
      setToast("El nombre no cambió");
      return;
    }
  
    try {
      await safService.rename(activeItem.uri, finalName);
      setRenameValue(finalName);
      setShowRename(false);
      setToast("Renombrado correctamente");
      await refreshCurrent();
    } catch (error: any) {
      setToast(error?.message || "No se pudo renombrar");
    }
  };

  const handleShare = async (item: FileEntry) => {
    try {
      await safService.shareFile(item.uri, item.mimeType);
    } catch (error: any) {
      setToast(error?.message || "No se pudo compartir");
    }
  };

  const handleCreateFolder = async (folderNameParam?: string) => {
    if (!currentUri) return;
  
    const finalName = (folderNameParam ?? newFolderName).trim();
    if (!finalName) return;
  
    try {
      await safService.createFolder(currentUri, finalName);
      setShowCreateFolder(false);
      setNewFolderName("");
      setToast("Carpeta creada");
      await refreshCurrent();
    } catch (error: any) {
      setToast(error?.message || "No se pudo crear la carpeta");
    }
  };

  const setClipboard = (item: FileEntry, mode: ClipboardMode) => {
    const clip = clipboardService.set(item, mode);
    setClipboardState(clip);
  };

  const clearClipboard = () => {
    clipboardService.clear();
    setClipboardState(null);
  };

  const pasteHere = async () => {
    if (isSpecialMode || !currentUri) {
      setToast("No se puede pegar aquí");
      return;
    }

    const clip = clipboardService.get();
    if (!clip) {
      setToast("Portapapeles vacío");
      return;
    }

    try {
      if (clip.mode === "copy") {
        await safService.copy(clip.item.uri, currentUri);
        setToast("Pegado correctamente");
      } else {
        await safService.move(clip.item.uri, currentUri);
        clipboardService.clear();
        setClipboardState(null);
        setToast("Movido correctamente");
      }

      await refreshCurrent();
    } catch (error: any) {
      setToast(error?.message || "No se pudo pegar");
    }
  };

  const handleMultiDelete = async () => {
    if (!selected.length) return;

    try {
      for (const item of selected) {
        await safService.delete(item.uri);
      }

      setSelected([]);
      setSelectionMode(false);
      setToast("Elementos eliminados");
      refreshCurrent();
    } catch {
      setToast("No se pudieron eliminar algunos elementos");
    }
  };

  const handleBreadcrumbNavigate = (crumb: Crumb, index: number) => {
    const next = breadcrumbs.slice(0, index + 1);
    setBreadcrumbs(next);
    loadDirectory(crumb.uri, crumb.name, false);
  };

  return (
    <IonPage>
      <IonHeader>
        <IonToolbar>
          <IonButton slot="start" fill="clear" onClick={handleBack}>
            <IonIcon icon={arrowBackOutline} />
          </IonButton>

          <IonTitle>
            {selectionMode
              ? `${selected.length} seleccionado(s)`
              : currentName}
          </IonTitle>

          {selectionMode ? (
            <IonButton
              slot="end"
              fill="clear"
              color="danger"
              onClick={() => {
                setSelectionMode(false);
                setSelected([]);
              }}
            >
              <IonIcon icon={closeOutline} />
            </IonButton>
          ) : (
            !isSpecialMode && (
              <IonButton slot="end" fill="clear" onClick={pasteHere}>
                <IonIcon icon={clipboardOutline} />
              </IonButton>
            )
          )}
        </IonToolbar>
      </IonHeader>

      <IonContent fullscreen>
        <IonRefresher
          slot="fixed"
          onIonRefresh={async (e) => {
            await refreshCurrent();
            e.detail.complete();
          }}
        >
          <IonRefresherContent />
        </IonRefresher>

        {!isSpecialMode && (
          <Breadcrumbs items={breadcrumbs} onNavigate={handleBreadcrumbNavigate} />
        )}

        <div className="p-2">
          <ActionBar
            search={search}
            sortBy={sortBy}
            filterBy={filterBy}
            viewMode={viewMode}
            selectionMode={selectionMode}
            onSearchChange={setSearch}
            onSortChange={setSortBy}
            onFilterChange={setFilterBy}
            onViewModeChange={(mode) => {
              setViewMode(mode);
              storageService.setViewMode(mode);
            }}
            onToggleSelectionMode={() => {
              setSelectionMode((prev) => !prev);
              setSelected([]);
            }}
          />
        </div>

        {clipboard && !selectionMode && !isSpecialMode && (
          <div className="px-4 pb-2 text-sm opacity-80">
            Portapapeles: <strong>{clipboard.item.name}</strong> ({clipboard.mode === "copy" ? "copiar" : "mover"})
          </div>
        )}

        {loading ? (
          <div className="p-6 text-center opacity-70">Cargando...</div>
        ) : visibleItems.length === 0 ? (
          <EmptyState
            title="No hay elementos"
            description={
              search
                ? "No se encontraron resultados para tu búsqueda."
                : "Esta carpeta no contiene archivos ni carpetas."
            }
          />
        ) : viewMode === "grid" ? (
          <IonGrid>
            <IonRow>
              {visibleItems.map((item) => (
                <IonCol size="6" sizeMd="4" sizeLg="3" key={item.uri}>
                  <FileItem
                    item={item}
                    selected={selected.some((x) => x.uri === item.uri)}
                    selectionMode={selectionMode}
                    viewMode="grid"
                    onClick={handleItemClick}
                    onToggleSelect={toggleSelect}
                    onToggleFavorite={toggleFavorite}
                    onOptions={handleOptions}
                  />
                </IonCol>
              ))}
            </IonRow>
          </IonGrid>
        ) : (
          <IonList inset>
            {visibleItems.map((item) => (
              <FileItem
                key={item.uri}
                item={item}
                selected={selected.some((x) => x.uri === item.uri)}
                selectionMode={selectionMode}
                viewMode="list"
                onClick={handleItemClick}
                onToggleSelect={toggleSelect}
                onToggleFavorite={toggleFavorite}
                onOptions={handleOptions}
              />
            ))}
          </IonList>
        )}

        {!isSpecialMode && !selectionMode && (
          <IonFab slot="fixed" vertical="bottom" horizontal="end">
            <IonFabButton onClick={() => setShowCreateFolder(true)}>
              <IonIcon icon={addOutline} />
            </IonFabButton>
          </IonFab>
        )}

        {selectionMode && selected.length > 0 && (
          <div className="fixed bottom-4 left-0 right-0 px-4 z-50">
            <div className="bg-white rounded-2xl shadow-lg p-3 flex justify-center gap-2">
              <IonButton
                fill="outline"
                onClick={() => {
                  if (selected.length === 1) {
                    setClipboard(selected[0], "copy");
                  }
                }}
              >
                <IonIcon icon={copyOutline} slot="start" />
                Copiar
              </IonButton>

              <IonButton
                fill="outline"
                onClick={() => {
                  if (selected.length === 1) {
                    setClipboard(selected[0], "cut");
                  }
                }}
              >
                <IonIcon icon={cutOutline} slot="start" />
                Mover
              </IonButton>

              <IonButton color="danger" onClick={handleMultiDelete}>
                <IonIcon icon={trashOutline} slot="start" />
                Eliminar
              </IonButton>
            </div>
          </div>
        )}

        <IonActionSheet
          cssClass="my-custom-action-sheet"
          isOpen={showActionSheet}
          onDidDismiss={() => setShowActionSheet(false)}
          header={activeItem?.name}
          buttons={[
            {
              text: activeItem?.favorite ? "Quitar de favoritos" : "Añadir a favoritos",
              icon: activeItem?.favorite ? star : starOutline,
              handler: () => activeItem && toggleFavorite(activeItem),
            },
            {
              text: "Copiar",
              icon: copyOutline,
              handler: () => activeItem && setClipboard(activeItem, "copy"),
            },
            {
              text: "Mover",
              icon: cutOutline,
              handler: () => activeItem && setClipboard(activeItem, "cut"),
            },
            {
              text: "Compartir",
              icon: shareOutline,
              handler: () => activeItem && handleShare(activeItem),
            },
            {
              text: "Renombrar",
              icon: createOutline,
              handler: () => setShowRename(true),
            },
            {
              text: "Propiedades",
              icon: informationCircleOutline,
              handler: () => setShowProperties(true),
            },
            {
              text: "Eliminar",
              role: "destructive",
              icon: trashOutline,
              handler: () => activeItem && handleDelete(activeItem),
            },
            {
              text: "Cancelar",
              role: "cancel",
            },
          ]}
        />

        <IonAlert
          isOpen={showCreateFolder}
          header="Nueva carpeta"
          inputs={[
            {
              name: "folderName",
              type: "text",
              placeholder: "Nombre de la carpeta",
              value: newFolderName,
            },
          ]}
          buttons={[
            {
              text: "Cancelar",
              role: "cancel",
              handler: () => {
                setShowCreateFolder(false);
                setNewFolderName("");
              },
            },
            {
              text: "Crear",
              handler: (data) => {
                const folderName = (data.folderName || "").trim();
                handleCreateFolder(folderName);
              },
            },
          ]}
          onDidDismiss={() => setShowCreateFolder(false)}
        />

        <IonAlert
          isOpen={showRename}
          header="Renombrar"
          inputs={[
            {
              name: "rename",
              type: "text",
              placeholder: "Nuevo nombre",
              value: renameValue,
            },
          ]}
          buttons={[
            {
              text: "Cancelar",
              role: "cancel",
              handler: () => setShowRename(false),
            },
            {
              text: "Guardar",
              handler: (data) => {
                const newName = (data.rename || "").trim();
                handleRename(newName);
              },
            },
          ]}
          onDidDismiss={() => setShowRename(false)}
        />

        <PropertiesModal
          isOpen={showProperties}
          item={activeItem}
          onClose={() => setShowProperties(false)}
        />

        <IonToast
          isOpen={!!toast}
          message={toast}
          duration={1800}
          onDidDismiss={() => setToast("")}
        />
      </IonContent>
    </IonPage>
  );
};

function extractNameFromUri(uri: string) {
  try {
    const decoded = decodeURIComponent(uri);
    const parts = decoded.split("/");
    return parts[parts.length - 1] || "Carpeta";
  } catch {
    return "Carpeta";
  }
}

export default Browser;
